def plot(df,type):
    
    A,B = curve_fit(f, df.Ncube.values,df.time.values)[0]


    x  = np.array([0,0.175*1e8])
    y = A*x+B
    
    plt.figure(dpi=100)
    plt.plot(x,y,label='f(x)')
    plt.plot(df.Ncube,df.time,'.', label='Measured time')
    plt.xlabel('$N^3$',fontsize = 15)
    plt.ylabel('time [$\mu s$]',fontsize = 15)
    plt.legend()
    plt.title("Matrix multiplication ("+type[0]+"-value, "+type[1]+"-value)")
    textstr = '\n'.join((r'f(x)=Ax+b',\
                                 r'$A = %.5f$' % A,\
                                 r'$B = %.2f$' % B))
    props = dict(boxstyle='round', alpha=0.5)
    plt.text(0.12*1e8,10000,s=textstr,fontsize=14,verticalalignment='top', bbox=props)
    plt.show()
